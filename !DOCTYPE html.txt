    animation-duration:${8+Math.random()*12}s;
    animation-delay:${-Math.random()*20}s;
    --drift:${(Math.random()-0.5)*120}px;
  `;
  pc.appendChild(p);
}

// ── DOTS ──────────────────────────────────────────
const dotsEl = document.getElementById('dots');
for (let i = 0; i < TOTAL_SCENES; i++) {
  const d = document.createElement('div');
  d.className = 'dot' + (i===0?' active':'');
  d.addEventListener('click', ()=>goTo(i));
  dotsEl.appendChild(d);
}

// ── PROGRESS ──────────────────────────────────────
function setProgress(n) {
  document.getElementById('progress').style.width = ((n+1)/TOTAL_SCENES*100)+'%';
}

// ── GOTO SCENE ────────────────────────────────────
function goTo(n) {
  if (animating || n === current) return;
  animating = true;

  const prev = document.getElementById('s'+(current+1));
  const next = document.getElementById('s'+(n+1));

  prev.classList.remove('active');
  setTimeout(()=>{
    next.classList.add('active');
    current = n;
    updateDots();
    setProgress(n);
    runSceneAnimation(n);
    setTimeout(()=>{ animating = false; }, 700);
  }, 400);

  resetAutoTimer();
}

function updateDots() {
  document.querySelectorAll('.dot').forEach((d,i)=>{
    d.classList.toggle('active', i===current);
  });
}

document.getElementById('nav-next').addEventListener('click', ()=>{
  if (current < TOTAL_SCENES-1) goTo(current+1);
});
document.getElementById('nav-prev').addEventListener('click', ()=>{
  if (current > 0) goTo(current-1);
});

// Keyboard
document.addEventListener('keydown', e=>{
  if (e.key==='ArrowRight'||e.key==='ArrowDown') { if(current<TOTAL_SCENES-1) goTo(current+1); }
  if (e.key==='ArrowLeft'||e.key==='ArrowUp')   { if(current>0) goTo(current-1); }
});

// ── AUTO ADVANCE ──────────────────────────────────
function resetAutoTimer() {
  clearTimeout(autoTimer);
  autoTimer = setTimeout(()=>{
    if (current < TOTAL_SCENES-1) goTo(current+1);
  }, 5000);
}

// ── SCENE ANIMATIONS ──────────────────────────────
function runSceneAnimation(n) {
  if (n === 1) animateStats();
  if (n === 2) animateMap();
  if (n === 3) animateSteps();
  if (n === 4) animateLayers();
  if (n === 5) animateBigNums();
  if (n === 6) animateFinal();
}

// Scene 2: counter stats
function animateStats() {
  document.querySelectorAll('.stat-num').forEach(el => {
    const target = +el.dataset.target;
    const suffix = el.dataset.suffix || '';
    let start = 0;
    const dur = 1400;
    const step = 16;
    const inc = target / (dur/step);
    const timer = setInterval(()=>{
      start = Math.min(start+inc, target);
      const val = target >= 1000 ? Math.round(start).toLocaleString() : Math.round(start*10)/10;
      el.textContent = val + suffix;
      if (start >= target) clearInterval(timer);
    }, step);
  });
}

// Scene 3: map clustering animation
function animateMap() {
  const vis = document.getElementById('mapVis');
  const svg = document.getElementById('connSvg');
  // Clear previous
  vis.querySelectorAll('.node,.cluster-circle').forEach(e=>e.remove());
  svg.innerHTML = '';

  const W = vis.offsetWidth, H = vis.offsetHeight;

  // Define clusters
  const clusters = [
    { cx: 0.25, cy: 0.35, r: 0.22, color: '#D94F1E22', stroke: '#D94F1E',
      nodes: [{x:.15,y:.2},{x:.28,y:.4},{x:.2,y:.55},{x:.35,y:.3}] },
    { cx: 0.72, cy: 0.5, r: 0.24, color: '#0A7EA415', stroke: '#0A7EA4',
      nodes: [{x:.6,y:.35},{x:.78,y:.28},{x:.82,y:.55},{x:.65,y:.62}] },
  ];
  const hub = { x:.5, y:.5 };

  // Draw hub
  addNode(vis, hub.x*W, hub.y*H, 'hub');

  // Animate cluster reveal
  clusters.forEach((cl, ci) => {
    // nodes first
    cl.nodes.forEach((nd, ni) => {
      setTimeout(()=>{
        const n = addNode(vis, nd.x*W, nd.y*H, 'pickup');
        n.style.opacity = '0';
        n.style.transition = 'opacity 0.4s';
        setTimeout(()=>{ n.style.opacity='1'; }, 50);
      }, ci*400 + ni*120);
    });

    // cluster circle
    setTimeout(()=>{
      const cc = document.createElement('div');
      cc.className = 'cluster-circle';
      cc.style.cssText = `
        left:${cl.cx*100}%; top:${cl.cy*100}%;
        width:${cl.r*2*W}px; height:${cl.r*2*H}px;
        background:${cl.color}; border-color:${cl.stroke};
        opacity:0; transition:opacity 0.6s;
      `;
      vis.appendChild(cc);
      setTimeout(()=>{ cc.style.opacity='1'; }, 50);

      // draw lines to hub
      cl.nodes.forEach(nd => {
        const line = document.createElementNS('http://www.w3.org/2000/svg','line');
        line.setAttribute('x1', nd.x*W); line.setAttribute('y1', nd.y*H);
        line.setAttribute('x2', hub.x*W); line.setAttribute('y2', hub.y*H);
        line.setAttribute('stroke', cl.stroke);
        line.setAttribute('stroke-width','1.5');
        line.setAttribute('stroke-dasharray','6 4');
        line.setAttribute('stroke-opacity','0.5');
        line.style.opacity='0';
        line.style.transition='opacity 0.5s';
        svg.appendChild(line);
        setTimeout(()=>{ line.style.opacity='1'; }, 200);
      });
    }, ci*400 + cl.nodes.length*120 + 200);
  });

  // drop nodes
  setTimeout(()=>{
    const drops = [{x:.45,y:.15},{x:.55,y:.8},{x:.85,y:.75}];
    drops.forEach((d,i)=>{
      setTimeout(()=>{
        const n = addNode(vis, d.x*W, d.y*H, 'drop');
        n.style.opacity='0'; n.style.transition='opacity 0.4s';
        setTimeout(()=>{ n.style.opacity='1'; }, 50);
        const line = document.createElementNS('http://www.w3.org/2000/svg','line');
        line.setAttribute('x1',hub.x*W); line.setAttribute('y1',hub.y*H);
        line.setAttribute('x2',d.x*W); line.setAttribute('y2',d.y*H);
        line.setAttribute('stroke','#1A3FCB');
        line.setAttribute('stroke-width','2');
        line.style.opacity='0'; line.style.transition='opacity 0.4s';
        svg.appendChild(line);
        setTimeout(()=>{ line.style.opacity='1'; },50);
      }, i*200);
    });
  }, 1800);
}

function addNode(parent, x, y, type) {
  const n = document.createElement('div');
  n.className = `node ${type}`;
  n.style.left = x+'px'; n.style.top = y+'px';
  parent.appendChild(n);
  return n;
}
